# SLAM 建图 Web 控制面板 — 实施计划 v2

> **原则:** 用户全程在浏览器操作，不碰终端
> **日期:** 2026-07-18
> **端口:** 5001（独立于 patrol_web :5000）

**目标:** 新建独立 Web 界面用于 SLAM 建图。slam_web 运行在 patrol_car (Foxy) 容器中常驻，建图时 Web 按钮一键启动 Humble 容器（slam_toolbox），地图数据通过跨容器 DDS 回传。

**架构:**
```
patrol_car 容器 (Foxy, 常驻)
├── slam_web (:5001)  ← 新建
│   ├── 手动控制 → /cmd_vel (本地)
│   ├── 激光显示 ← /scan (本地)
│   ├── 地图显示 ← /map (跨容器 DDS, 来自 Humble)
│   └── /api/slam/start → docker compose up slam_nav
│   └── /api/slam/stop  → docker compose stop slam_nav
│   └── /api/save_map   → ros2 run map_saver_cli
│
slam_nav 容器 (Humble, 按需启动)
└── slam_toolbox → /map ──跨容器 DDS──→ patrol_car

操作流程:
  浏览器打开 :5001 → 看到「SLAM 未启动」→ 点击「开始建图」
  → Web 触发 docker compose up slam_nav → 5秒后 /map 有数据
  → 地图实时显示 → 方向键控制建图 → 点击「保存地图」
  → 点「停止建图」关闭 Humble 容器
```

## Task 2: 后端 web_server.py (Flask + ROS2)

**文件:** 新建 `src/slam_web/slam_web/web_server.py`

**关键设计:**
- Flask 端口 5001
- 订阅 /map (跨容器 DDS, 来自 Humble) → 转 base64 PNG
- 订阅 /scan (本地) → 激光可视化
- 发布 /cmd_vel (本地) → 底盘控制
- `/api/slam/start` → `docker compose -f /home/pi/patrol_robot/docker-compose.yml up -d slam_nav`
- `/api/slam/stop` → `docker compose -f /home/pi/patrol_robot/docker-compose.yml stop slam_nav`
- `/api/save_map` → `ros2 run nav2_map_server map_saver_cli -f ~/maps/<name>`
- 需要 `pip3 install pillow`（OccupancyGrid → PNG）
- 需要 `pip3 install docker` 或直接用 subprocess 调 docker 命令

```python
#!/usr/bin/env python3
"""SLAM 建图 Web — patrol_car 容器内运行, 端口 5001"""
import json, time, threading, subprocess, base64, io
from pathlib import Path
from datetime import datetime
import rclpy
from rclpy.node import Node
from nav_msgs.msg import OccupancyGrid
from geometry_msgs.msg import Twist
from sensor_msgs.msg import LaserScan
import numpy as np
from flask import Flask, render_template, jsonify, request
from PIL import Image as PILImage

app = Flask(__name__)
_store = {
    "map_png": None, "map_info": {}, "scan_ranges": [],
    "slam_status": "stopped",  # stopped | starting | mapping | saving
    "cmd_vel_pub": None,
}

@app.route("/")
def index():
    return render_template("slam.html")

@app.route("/api/map")
def api_map():
    return jsonify({"png": _store["map_png"], "info": _store["map_info"], "status": _store["slam_status"]})

@app.route("/api/control", methods=["POST"])
def api_control():
    data = request.get_json() or {}
    d, dur, spd = data.get("direction","stop"), min(float(data.get("duration",0.3)),2.0), float(data.get("speed",0.2))
    pub = _store.get("cmd_vel_pub")
    if not pub: return jsonify({"error":"not ready"}), 503
    t = Twist()
    if d == "forward": t.linear.x = spd
    elif d == "backward": t.linear.x = -spd
    elif d == "left": t.angular.z = spd * 3
    elif d == "right": t.angular.z = -spd * 3
    pub.publish(t)
    def _stop():
        time.sleep(dur); pub.publish(Twist())
    threading.Thread(target=_stop, daemon=True).start()
    return jsonify({"ok":True, "direction":d})

@app.route("/api/slam/start", methods=["POST"])
def api_slam_start():
    _store["slam_status"] = "starting"
    try:
        subprocess.run(
            ["docker", "compose", "-f", "/home/pi/patrol_robot/docker-compose.yml", "up", "-d", "slam_nav"],
            capture_output=True, text=True, timeout=30
        )
        _store["slam_status"] = "mapping"
        return jsonify({"ok":True, "status":"mapping"})
    except Exception as e:
        _store["slam_status"] = "stopped"
        return jsonify({"error":str(e)}), 500

@app.route("/api/slam/stop", methods=["POST"])
def api_slam_stop():
    try:
        subprocess.run(
            ["docker", "compose", "-f", "/home/pi/patrol_robot/docker-compose.yml", "stop", "slam_nav"],
            capture_output=True, text=True, timeout=15
        )
        _store["slam_status"] = "stopped"
        _store["map_png"] = None
        return jsonify({"ok":True})
    except Exception as e:
        return jsonify({"error":str(e)}), 500

@app.route("/api/save_map", methods=["POST"])
def api_save_map():
    data = request.get_json() or {}
    name = data.get("name", datetime.now().strftime("map_%Y%m%d_%H%M%S"))
    save_dir = str(Path.home() / "maps")
    Path(save_dir).mkdir(parents=True, exist_ok=True)
    _store["slam_status"] = "saving"
    try:
        result = subprocess.run(
            ["ros2", "run", "nav2_map_server", "map_saver_cli", "-f", f"{save_dir}/{name}"],
            capture_output=True, text=True, timeout=30
        )
        _store["slam_status"] = "mapping"
        return jsonify({"ok":True, "name":name, "path":f"{save_dir}/{name}"})
    except Exception as e:
        _store["slam_status"] = "mapping"
        return jsonify({"error":str(e)}), 500

class SlamWebNode(Node):
    def __init__(self):
        super().__init__("slam_web")
        self.map_sub = self.create_subscription(OccupancyGrid, "/map", self._on_map, 10)
        self.scan_sub = self.create_subscription(LaserScan, "/scan", self._on_scan, 10)
        self.cmd_vel_pub = self.create_publisher(Twist, "/cmd_vel", 10)
        _store["cmd_vel_pub"] = self.cmd_vel_pub
        self.get_logger().info("SLAM Web :5001 ready")

    def _on_map(self, msg):
        try:
            w, h = msg.info.width, msg.info.height
            data = np.array(msg.data, dtype=np.int8).reshape(h, w)
            img = np.full((h, w), 128, dtype=np.uint8)  # unknown=128 gray
            img[data == 0] = 255   # free=white
            img[data > 0] = 0      # occupied=black
            pil = PILImage.fromarray(img, mode="L")
            buf = io.BytesIO(); pil.save(buf, format="PNG")
            _store["map_png"] = base64.b64encode(buf.getvalue()).decode()
            _store["map_info"] = {"width":w,"height":h,"resolution":round(msg.info.resolution,3)}
        except Exception as e:
            self.get_logger().error(f"map render: {e}")

    def _on_scan(self, msg):
        _store["scan_ranges"] = list(msg.ranges)[-360:]

def main():
    rclpy.init()
    node = SlamWebNode()
    app.template_folder = str(Path(__file__).parent.parent / "templates")
    threading.Thread(target=app.run, kwargs={"host":"0.0.0.0","port":5001,"debug":False,"use_reloader":False}, daemon=True).start()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node(); rclpy.shutdown()

if __name__ == "__main__":
    main()
```

## Task 8: 建图模式切换 — 停/启 patrol 节点

**目标:** 开始建图时自动停掉非必要节点(~660MB)，建图完成后恢复

**实现方式:** slam_web 维护一个进程名→启动命令映射表，通过 kill + subprocess 管理

```python
# 在 web_server.py 中增加:

# 节点管理: 名称 → (进程匹配模式, 重启命令)
PATROL_NODES = {
    "patrol_yolo":     ("yolo_detector", "ros2 run patrol_yolo yolo_detector --ros-args -p model_path:=/home/pi/patrol_robot/models/yolov5n.onnx -p confidence:=0.5 -p detect_interval:=1.0 > /tmp/yolo.log 2>&1 &"),
    "patrol_web":      ("web_server",    "ros2 run patrol_web web_server > /tmp/web.log 2>&1 &"),
    "alert_dispatcher":("alert_dispatcher", "ros2 run patrol_alert alert_dispatcher > /tmp/alert.log 2>&1 &"),
    "patrol_voice":    ("voice_node",    "ros2 run patrol_voice voice_node --ros-args -p serial_port:=/dev/myspeech -p voice_enabled:=true -p command_enabled:=true > /tmp/voice.log 2>&1 &"),
    "patrol_state":    ("patrol_state_machine", "ros2 run patrol_manager patrol_state_machine > /tmp/state.log 2>&1 &"),
}

def _patrol_stop():
    """停止所有巡逻节点, 释放内存"""
    import os, signal
    killed = []
    for name, (pattern, _) in PATROL_NODES.items():
        # 查找进程 PID
        result = subprocess.run(["pgrep", "-f", pattern], capture_output=True, text=True)
        for pid in result.stdout.strip().split():
            try:
                os.kill(int(pid), signal.SIGTERM)
                killed.append(name)
            except: pass
    return killed

def _patrol_start():
    """重新启动巡逻节点"""
    for name, (_, cmd) in PATROL_NODES.items():
        subprocess.Popen(cmd, shell=True, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
```

**API:**
- `POST /api/slam/start` → 先 `_patrol_stop()` → 再 `docker compose up slam_nav`
- `POST /api/slam/stop` → 先 `docker compose stop slam_nav` → 再 `_patrol_start()`

**前端:** 开始建图时弹提示「正在释放巡逻节点...」
